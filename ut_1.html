Cypress Query Examples

1. cy.get(selector)

// Select DOM elements by CSS selector
cy.get('.btn');
cy.get('#submit-btn');
cy.get('[data-testid="username-input"]');

2. cy.contains(content)

// Find elements containing specific text
cy.contains('Submit');
cy.contains('Special Item');

// Scoped within an element
cy.contains('.btn', 'Save');

3. cy.find(selector)

// Find descendant elements within a parent
cy.get('.button-group').find('.btn');
cy.get('.nested-container').find('.target-text');

4. cy.root()

// Get the root DOM element
cy.root();
// Typically used with .within()
cy.get('form').within(() => {
  cy.root().should('exist');
});
